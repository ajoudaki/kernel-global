\documentclass{standalone}
\usepackage{tikz}
\usepackage{tikz-3dplot}

\begin{document}

\tdplotsetmaincoords{70}{110}

\begin{tikzpicture}[tdplot_main_coords, scale=3]
    % Define the vectors
    \def\vOneX{0.8}  \def\vOneY{0.3}  \def\vOneZ{0.5}
    \def\vTwoX{-0.4} \def\vTwoY{0.7}  \def\vTwoZ{0.6}
    \def\vThreeX{0.5} \def\vThreeY{-0.6} \def\vThreeZ{0.7}

    % Draw the coordinate system
    \draw[thick,->] (0,0,0) -- (1,0,0) node[anchor=north east]{$x$};
    \draw[thick,->] (0,0,0) -- (0,1,0) node[anchor=north west]{$y$};
    \draw[thick,->] (0,0,0) -- (0,0,1) node[anchor=south]{$z$};

    % Draw the vectors
    \draw[red,thick,->] (0,0,0) -- (\vOneX,\vOneY,\vOneZ) node[anchor=south]{$\vec{v}_1$};
    \draw[green,thick,->] (0,0,0) -- (\vTwoX,\vTwoY,\vTwoZ) node[anchor=north]{$\vec{v}_2$};
    \draw[blue,thick,->] (0,0,0) -- (\vThreeX,\vThreeY,\vThreeZ) node[anchor=south]{$\vec{v}_3$};

    % Draw projections on the xy-plane
    \draw[red,dashed] (\vOneX,\vOneY,\vOneZ) -- (\vOneX,\vOneY,0);
    \draw[green,dashed] (\vTwoX,\vTwoY,\vTwoZ) -- (\vTwoX,\vTwoY,0);
    \draw[blue,dashed] (\vThreeX,\vThreeY,\vThreeZ) -- (\vThreeX,\vThreeY,0);

    % Label the vectors
    \node[red] at (0.5*\vOneX,0.5*\vOneY,0.5*\vOneZ) {$\vec{v}_1$};
    \node[green] at (0.5*\vTwoX,0.5*\vTwoY,0.5*\vTwoZ) {$\vec{v}_2$};
    \node[blue] at (0.5*\vThreeX,0.5*\vThreeY,0.5*\vThreeZ) {$\vec{v}_3$};
\end{tikzpicture}

\end{document}
